CREATE TABLE `story` (
  `title` varchar(100) NOT NULL,
  `authorUsername` varchar(100) NOT NULL,
  `genre` varchar(20) NOT NULL,
  `coverPage` varchar(255) NOT NULL,
  `lastEditDate` date NOT NULL DEFAULT current_timestamp(),
  `description` varchar(255) NOT NULL,
  `rating` varchar(20) NOT NULL,
  `completed` tinyint(1) DEFAULT NULL,
  `noOfChapters` int(3) DEFAULT NULL,
  `noOfReads` int(10) DEFAULT NULL,
  `chapterNumber` int(4) NOT NULL DEFAULT 1,
  `text` longtext DEFAULT NULL,
  PRIMARY KEY (`title`,`authorUsername`,`chapterNumber`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
